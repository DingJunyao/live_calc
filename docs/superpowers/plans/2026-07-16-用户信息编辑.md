# 用户信息编辑 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans（本环境 subagent 不可用，走 inline 执行）。Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **项目约束（CLAUDE.md）：** 不自行 `git commit`/分支（由用户执行）；不在对话中启动服务（前后端自动重载已开）；后端改动确保无语法错误（py_compile + pytest）；前端改动确保 `npm run build` 通过。本功能无表结构变更，无需 alembic / SQL 脚本。

**Goal:** 让普通用户在个人中心自助修改用户名、邮箱、手机号、密码（改密码校验旧密、改成功无缝续登）。

**Architecture:** 后端新增 `PUT /auth/me/account`（字段全可选，传了才改）+ 2 个 schema；改密码走 `verify_password` 校验旧密 → `token_version+=1` → 签发新 token 返回。前端 ProfileView 加「用户信息编辑」入口 + 单对话框（基本信息区 + 修改密码区），userStore 加 `setTokens` 供改密续登复用。

**Tech Stack:** FastAPI + SQLAlchemy + python-jose（后端）；Vue 3 + Vuetify 3 + Pinia（前端）；bcrypt + 前端 SHA256（`@/utils/crypto` 的 `hashPassword`）。

**Spec:** [docs/superpowers/specs/2026-07-16-用户信息编辑-design.md](../specs/2026-07-16-用户信息编辑-design.md)

```python
@router.put("/me/account", response_model=UserAccountResponse)
async def update_my_account(
    payload: UserAccountUpdate,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db),
):
    """当前用户自行更新账号信息（用户名/邮箱/手机/密码）。

    - 用户名/邮箱/手机：传了且与当前不同才改，查重（排除自己），命中 400。
    - 密码：new_password 非空时必须校验 current_password；成功后 token_version+1
      并签发新 token 返回（发起方无缝续登，其他设备旧 token 失效）。
      不改密码时 token_version 不动，不签发 token。
    """
    from app.core.security import (
        verify_password, get_password_hash, create_access_token, create_refresh_token,
    )

    # current_user 来自 get_current_user 的独立 session（detached），
    # 在本请求 db 内重新加载，保证 setattr + commit + refresh 生效（与 patch_me 一致）。
    user = db.query(User).filter(User.id == current_user.id).first()
    if not user:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="用户不存在")

    # 用户名
    if payload.username is not None and payload.username != user.username:
        if db.query(User).filter(
            User.username == payload.username, User.id != user.id
        ).first():
            raise HTTPException(status_code=400, detail="用户名已被占用")
        user.username = payload.username

    # 邮箱
    if payload.email is not None and payload.email != user.email:
        if db.query(User).filter(
            User.email == payload.email, User.id != user.id
        ).first():
            raise HTTPException(status_code=400, detail="邮箱已被占用")
        user.email = payload.email

    # 手机（phone 允许 NULL，查重时 NULL 不算冲突）
    if payload.phone is not None and payload.phone != user.phone:
        if payload.phone and db.query(User).filter(
            User.phone == payload.phone, User.id != user.id
        ).first():
            raise HTTPException(status_code=400, detail="手机号已被占用")
        user.phone = payload.phone

    # 密码
    changed_password = False
    if payload.new_password is not None:
        if not payload.current_password:
            raise HTTPException(status_code=400, detail="修改密码需提供当前密码")
        if not verify_password(payload.current_password, user.password_hash):
            raise HTTPException(status_code=401, detail="当前密码错误")
        user.password_hash = get_password_hash(payload.new_password)
        user.token_version += 1
        changed_password = True

    db.commit()
    db.refresh(user)

    access_token = None
    refresh_token = None
    if changed_password:
        token_data = {"sub": str(user.id), "ver": user.token_version}
        access_token = create_access_token(token_data)
        refresh_token = create_refresh_token(token_data)

    return UserAccountResponse(
        user=_user_to_response(user, db),
        access_token=access_token,
        refresh_token=refresh_token,
    )
```

```ts
function openAccountDialog() {
  const u = userStore.user as any
  accountForm.value = {
    username: u?.username ?? '',
    email: u?.email ?? '',
    phone: u?.phone ?? '',
    currentPassword: '',
    newPassword: '',
    confirmPassword: '',
  }
  Object.keys(accountErrors).forEach(k => (accountErrors[k as keyof typeof accountErrors] = ''))
  accountDialog.value = true
}

async function saveAccount() {
  Object.keys(accountErrors).forEach(k => (accountErrors[k as keyof typeof accountErrors] = ''))
  const f = accountForm.value
  let hasError = false
  if (!f.username || f.username.length < 3 || f.username.length > 50) {
    accountErrors.username = '用户名需 3-50 个字符'; hasError = true
  }
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(f.email)) {
    accountErrors.email = '邮箱格式不正确'; hasError = true
  }
  if (f.phone && !/^1[3-9]\d{9}$/.test(f.phone)) {
    accountErrors.phone = '手机号格式不正确'; hasError = true
  }
  const changingPassword = !!(f.newPassword || f.currentPassword || f.confirmPassword)
  if (changingPassword) {
    if (!f.currentPassword) { accountErrors.currentPassword = '请输入当前密码'; hasError = true }
    if (!f.newPassword || f.newPassword.length < 6) {
      accountErrors.newPassword = '新密码至少 6 个字符'; hasError = true
    }
    if (f.newPassword !== f.confirmPassword) {
      accountErrors.confirmPassword = '两次输入的新密码不一致'; hasError = true
    }
  }
  if (hasError) return

  savingAccount.value = true
  try {
    const u = userStore.user as any
    const payload: Record<string, any> = {}
    if (f.username !== u?.username) payload.username = f.username
    if (f.email !== u?.email) payload.email = f.email
    if ((f.phone || null) !== (u?.phone || null)) payload.phone = f.phone || null
    if (changingPassword) {
      payload.current_password = hashPassword(f.currentPassword)
      payload.new_password = hashPassword(f.newPassword)
    }
    if (Object.keys(payload).length === 0) {
      accountDialog.value = false
      return
    }
    const data: any = await api.put('/auth/me/account', payload)
    if (data.access_token && data.refresh_token) {
      userStore.setTokens(data.access_token, data.refresh_token)
    }
    await userStore.fetchUser()
    accountDialog.value = false
    notify(changingPassword ? '已更新，登录态已刷新' : '已更新', 'success')
  } catch (e: any) {
    notify('保存失败：' + (e?.userMessage || e?.message || '未知错误'), 'error')
  } finally {
    savingAccount.value = false
  }
}
```
